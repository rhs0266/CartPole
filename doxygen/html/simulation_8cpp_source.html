66  double desiredVelocity = -1.3; // z-axis
67  double currentVelocity = world->GetCharacter()->GetSkeleton()->getVelocity(5);
68  double zVelocityReward = exp(-pow((desiredVelocity - currentVelocity),2.0));

71  double torsoUpwardReward = exp(-pow(torsoAngle, 2.0));
72

74  double fallDownPenalty = fallDownFlag ? -10 : 0.0;
75

84  }
85  }
